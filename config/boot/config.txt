# uConsole (CM4) - Raspberry Pi firmware boot config
# Base: Arch Linux ARM rpi-aarch64
#
# devterm-* の各 overlay と devterm-panel-uc(DSIパネル) は ClockworkPi の
# パッチ済みカーネルに含まれるものです。scripts/build-kernel.sh でビルドした
# カーネル成果物を導入して初めて有効になります（README 参照）。
# ベースは ClockworkPi 公式 uConsole CM4 の config.txt に準拠。

disable_overscan=1
dtparam=audio=on

[pi4]
max_framebuffers=2

[all]
# 内蔵LCD自動検出を無効化し、uConsole 専用パネルを使う
ignore_lcd=1

# USB (CM4 dwc2 を host モードで)
dtoverlay=dwc2,dr_mode=host

# KMS / GPU (uConsole は pi4 版 + CMA 384MB)
dtoverlay=vc4-kms-v3d-pi4,cma-384

# --- uConsole 専用 overlay（パッチ済みカーネルが提供） ---
dtoverlay=devterm-pmu
dtoverlay=devterm-panel-uc
dtoverlay=devterm-misc

# オーディオ (GPIO12/13 PWM)
dtoverlay=audremap,pins_12_13

# SPI / アンテナ等
dtparam=spi=on
gpio=10=ip,np
dtparam=ant2
